Every Mintlify site needs a `docs.json` file with core configuration settings. This file controls everything from branding to navigation structure.

<Note>
The `docs.json` file should be placed in the root of your documentation directory.
</Note>

## Basic Configuration

Start with these essential properties to get your documentation site up and running.

<ResponseField name="name" type="string" required>
  Name of your project. Used for the global title and SEO.
  
  ```json
  "name": "Mint Starter Kit"
  ```
</ResponseField>

<ResponseField name="$schema" type="string">
  URL to the Mintlify docs.json schema for editor autocomplete and validation.
  
  ```json
  "$schema": "https://mintlify.com/docs.json"
  ```
</ResponseField>

## Branding

### Logo

<ResponseField name="logo" type="string | object">
  Path to logo image or object with paths for light and dark mode logos.
  
  <Expandable title="Logo Configuration">
    <ResponseField name="light" type="string">
      Path to the logo in light mode
      
      ```json
      "light": "/logo/light.svg"
      ```
    </ResponseField>
    
    <ResponseField name="dark" type="string">
      Path to the logo in dark mode
      
      ```json
      "dark": "/logo/dark.svg"
      ```
    </ResponseField>
    
    <ResponseField name="href" type="string" default="/">
      Where clicking on the logo links you to
    </ResponseField>
  </Expandable>
</ResponseField>

**Example:**
```json
"logo": {
  "light": "/logo/light.svg",
  "dark": "/logo/dark.svg"
}
```

### Favicon

<ResponseField name="favicon" type="string">
  Path to the favicon image displayed in browser tabs.
  
  ```json
  "favicon": "/favicon.svg"
  ```
</ResponseField>

### Colors

<ResponseField name="colors" type="object" required>
  Hex color codes for your global theme.
  
  <Expandable title="Color Properties">
    <ResponseField name="primary" type="string" required>
      The primary color used for highlighted content, section headers, and accents in light mode.
      
      ```json
      "primary": "#16A34A"
      ```
    </ResponseField>
    
    <ResponseField name="light" type="string">
      The primary color for dark mode. Used for highlighted content and accents.
      
      ```json
      "light": "#07C983"
      ```
    </ResponseField>
    
    <ResponseField name="dark" type="string">
      The primary color for important buttons and call-to-action elements.
      
      ```json
      "dark": "#15803D"
      ```
    </ResponseField>
    
    <ResponseField name="background" type="object">
      The background color in both light and dark mode.
      
      <Expandable title="Background Colors">
        <ResponseField name="light" type="string" required>
          Hex color code of the background in light mode
        </ResponseField>
        <ResponseField name="dark" type="string" required>
          Hex color code of the background in dark mode
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

**Example:**
```json
"colors": {
  "primary": "#16A34A",
  "light": "#07C983",
  "dark": "#15803D"
}
```

### Theme

<ResponseField name="theme" type="string" default="prism">
  The color theme for your documentation. Options include `prism`, `mint`, `quill`, and more.
  
  ```json
  "theme": "mint"
  ```
</ResponseField>

## Navigation Structure

<ResponseField name="navigation" type="object" required>
  Defines the structure of your documentation navigation.
  
  <Expandable title="Navigation Configuration">
    <ResponseField name="tabs" type="Tab[]">
      Array of top-level tabs in your navigation.
      
      <Expandable title="Tab Properties">
        <ResponseField name="tab" type="string">
          The name of the tab
        </ResponseField>
        <ResponseField name="groups" type="Group[]">
          Array of navigation groups within this tab
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="global" type="object">
      Global navigation elements like anchors that appear across all tabs.
      
      <Expandable title="Global Properties">
        <ResponseField name="anchors" type="Anchor[]">
          Array of external links displayed in the navigation sidebar.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

See the [Navigation](/essentials/navigation) page for detailed navigation configuration.

## Topbar Configuration

### Navbar Links

<ResponseField name="navbar" type="object">
  Configuration for the top navigation bar.
  
  <Expandable title="Navbar Configuration">
    <ResponseField name="links" type="Link[]">
      Array of links displayed in the navbar.
      
      ```json
      "links": [
        {
          "label": "Support",
          "href": "mailto:hi@mintlify.com"
        }
      ]
      ```
    </ResponseField>
    
    <ResponseField name="primary" type="object">
      Primary call-to-action button in the navbar.
      
      <Expandable title="Primary Button">
        <ResponseField name="type" type="string">
          Button type: `button` or `link`
        </ResponseField>
        <ResponseField name="label" type="string">
          Text displayed on the button
        </ResponseField>
        <ResponseField name="href" type="string">
          URL the button links to
        </ResponseField>
      </Expandable>
      
      ```json
      "primary": {
        "type": "button",
        "label": "Dashboard",
        "href": "https://dashboard.mintlify.com"
      }
      ```
    </ResponseField>
  </Expandable>
</ResponseField>

### Legacy Topbar (Deprecated)

<Warning>
The `topbarLinks` and `topbarCtaButton` fields are deprecated. Use `navbar` instead.
</Warning>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  Array of links in the topbar (legacy).
  
  <Expandable title="TopbarLink">
    <ResponseField name="name" type="string">
      The name of the button
    </ResponseField>
    <ResponseField name="url" type="string">
      The URL once you click on the button
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="object">
  Call-to-action button in the topbar (legacy).
  
  <Expandable title="Topbar Call to Action">
    <ResponseField name="type" type="'link' | 'github'" default="link">
      Link shows a button. GitHub shows repo information including star count.
    </ResponseField>
    <ResponseField name="url" type="string">
      Button URL or GitHub repository URL
    </ResponseField>
    <ResponseField name="name" type="string">
      Text inside the button (only for `link` type)
    </ResponseField>
  </Expandable>
</ResponseField>

## Footer Configuration

<ResponseField name="footer" type="object">
  Configuration for the footer section.
  
  <Expandable title="Footer Configuration">
    <ResponseField name="socials" type="object">
      Social media links displayed in the footer.
      
      ```json
      "socials": {
        "x": "https://x.com/mintlify",
        "github": "https://github.com/mintlify",
        "linkedin": "https://linkedin.com/company/mintlify"
      }
      ```
      
      Supported platforms: `website`, `facebook`, `x`, `discord`, `slack`, `github`, `linkedin`, `instagram`, `hacker-news`
    </ResponseField>
  </Expandable>
</ResponseField>

### Legacy Footer (Deprecated)

<ResponseField name="footerSocials" type="object">
  Legacy footer socials configuration. Use `footer.socials` instead.
  
  ```json
  "footerSocials": {
    "x": "https://x.com/mintlify",
    "website": "https://mintlify.com"
  }
  ```
</ResponseField>

## API Configuration

<ResponseField name="api" type="object">
  Configuration for API documentation and playground.
  
  <Expandable title="API Configuration">
    <ResponseField name="baseUrl" type="string | string[]">
      The base URL for all API endpoints. If an array, enables multiple base URL options.
      
      ```json
      "baseUrl": "https://api.example.com"
      ```
    </ResponseField>
    
    <ResponseField name="auth" type="object">
      Authentication configuration for API playground.
      
      <Expandable title="Auth Configuration">
        <ResponseField name="method" type="'bearer' | 'basic' | 'key'">
          The authentication strategy used for all API endpoints
        </ResponseField>
        <ResponseField name="name" type="string">
          Name of the authentication parameter. For `basic` auth, use format: `[usernameName]:[passwordName]`
        </ResponseField>
        <ResponseField name="inputPrefix" type="string">
          Default prefix for the authentication input field
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="playground" type="object">
      Configurations for the API playground.
      
      <Expandable title="Playground Configuration">
        <ResponseField name="mode" type="'show' | 'simple' | 'hide'" default="show">
          Whether the playground is showing, hidden, or displaying only the endpoint with no interactivity
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="maintainOrder" type="boolean">
      Ensures key ordering in OpenAPI pages matches the OpenAPI file definition.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="openapi" type="string | string[]">
  URL(s) or relative path(s) pointing to your OpenAPI file.
  
  ```json
  "openapi": "https://example.com/openapi.json"
  ```
  
  Or with multiple files:
  ```json
  "openapi": [
    "https://example.com/openapi1.json",
    "/openapi2.json"
  ]
  ```
</ResponseField>

## Advanced Features

### Anchors

<ResponseField name="anchors" type="Anchor[]">
  Array of external navigation anchors (legacy - use `navigation.global.anchors` instead).
  
  <Expandable title="Anchor Properties">
    <ResponseField name="icon" type="string">
      Font Awesome icon name
      
      ```json
      "icon": "book-open-cover"
      ```
    </ResponseField>
    
    <ResponseField name="name" type="string">
      The anchor label
    </ResponseField>
    
    <ResponseField name="url" type="string">
      The URL the anchor links to
    </ResponseField>
    
    <ResponseField name="color" type="string">
      Hex color or gradient object with `from` and `to` properties
    </ResponseField>
    
    <ResponseField name="iconType" type="string" default="duotone">
      One of: `brands`, `duotone`, `light`, `sharp-solid`, `solid`, or `thin`
    </ResponseField>
  </Expandable>
</ResponseField>

### Tabs

<ResponseField name="tabs" type="Tab[]">
  Array of navigational tabs (legacy - use `navigation.tabs` instead).
  
  <Expandable title="Tab Properties">
    <ResponseField name="name" type="string">
      The name of the tab label
    </ResponseField>
    <ResponseField name="url" type="string">
      The URL path that marks what pages go in this tab
    </ResponseField>
  </Expandable>
</ResponseField>

### Versions

<ResponseField name="versions" type="string[]">
  Array of version names for multi-version documentation.
  
  ```json
  "versions": ["v2.0", "v1.0"]
  ```
</ResponseField>

### Mode Toggle

<ResponseField name="modeToggle" type="object">
  Customize the dark/light mode toggle.
  
  <Expandable title="ModeToggle Configuration">
    <ResponseField name="default" type="'light' | 'dark'">
      Set default mode for new users. Otherwise defaults to system preference.
    </ResponseField>
    
    <ResponseField name="isHidden" type="boolean" default="false">
      Hide the dark/light mode toggle. Combine with `default` to force a single mode.
    </ResponseField>
  </Expandable>
</ResponseField>

**Force dark mode only:**
```json
"modeToggle": {
  "default": "dark",
  "isHidden": true
}
```

### Feedback Options

<ResponseField name="feedback" type="object">
  Enable feedback buttons on documentation pages.
  
  <Expandable title="Feedback Configuration">
    <ResponseField name="suggestEdit" type="boolean" default="false">
      Enables a button to allow users to suggest edits via pull requests
    </ResponseField>
    
    <ResponseField name="raiseIssue" type="boolean" default="false">
      Enables a button to allow users to raise an issue about the documentation
    </ResponseField>
  </Expandable>
</ResponseField>

### Contextual Actions

<ResponseField name="contextual" type="object">
  Configure contextual actions available on documentation pages.
  
  <Expandable title="Contextual Configuration">
    <ResponseField name="options" type="string[]">
      Array of enabled contextual actions.
      
      ```json
      "options": [
        "copy",
        "view",
        "chatgpt",
        "claude",
        "perplexity",
        "mcp",
        "cursor",
        "vscode"
      ]
      ```
    </ResponseField>
  </Expandable>
</ResponseField>

### Background Image

<ResponseField name="backgroundImage" type="string">
  Path to a background image displayed behind every page.
  
  ```json
  "backgroundImage": "/images/background.png"
  ```
</ResponseField>

## Complete Example

Here's a complete `docs.json` example combining all major configuration options:

```json
{
  "$schema": "https://mintlify.com/docs.json",
  "theme": "mint",
  "name": "Mint Starter Kit",
  "colors": {
    "primary": "#16A34A",
    "light": "#07C983",
    "dark": "#15803D"
  },
  "favicon": "/favicon.svg",
  "logo": {
    "light": "/logo/light.svg",
    "dark": "/logo/dark.svg"
  },
  "navbar": {
    "links": [
      {
        "label": "Support",
        "href": "mailto:hi@mintlify.com"
      }
    ],
    "primary": {
      "type": "button",
      "label": "Dashboard",
      "href": "https://dashboard.mintlify.com"
    }
  },
  "footer": {
    "socials": {
      "x": "https://x.com/mintlify",
      "github": "https://github.com/mintlify",
      "linkedin": "https://linkedin.com/company/mintlify"
    }
  },
  "navigation": {
    "tabs": [
      {
        "tab": "Guides",
        "groups": [
          {
            "group": "Getting started",
            "pages": ["index", "quickstart"]
          }
        ]
      }
    ],
    "global": {
      "anchors": [
        {
          "anchor": "Documentation",
          "href": "https://mintlify.com/docs",
          "icon": "book-open-cover"
        }
      ]
    }
  },
  "contextual": {
    "options": ["copy", "view", "chatgpt", "cursor"]
  }
}
```
